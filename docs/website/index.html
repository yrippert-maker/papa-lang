Compiled Python

from papa_lang.orchestrator import Orchestrator
from papa_lang.swarm import SwarmRunner, SwarmAgent
from papa_lang.hrs import HRSConfig
from papa_guard import Guard

synthesis = SwarmAgent(
    name='synthesis',
    model='gemini-1.5-pro',
    guard=Guard(pii_enabled=True),
    hrs_config=HRSConfig(threshold=0.1),
    memory_enabled=True,
)

medical_team = SwarmRunner(
    name='medical_team',
    agents=[synthesis, research],
    consensus=ConsensusConfig(required=4, of=7),
    anchor='blockchain',
    pii_filter=True,
    hrs_max=0.15,
)

orchestrator = Orchestrator(
    route='orchestrator',
    fallback='single',
    module='papa-life',
)
